 86@_socketio_server.event
 87async def connect(sid, environment):
 88	"""
 89	This func is called by the socketio server when a new client connects.
 90	It will trigger an endpoint to trigger the corresponding inner function in the update loop.
 91	This complex workaround is necessary, because the socketio server is not part of the fastapi app (it is just mounted
 92	to the path), so it cannot access the FastAPI's dependency to the system task queue function to trigger logic within
 93	the update loop.
 94
 95	:param sid: socket id
 96	:param environment: environment dict
 97	:return:
 98	"""
 99
100	print("connect")
101	global origin_url
102	origin_url = environment["HTTP_ORIGIN"]
103	endpoint_url = origin_url + "/api/v1/system/data_subscription/connect_client"
104	# perform api request with httpx
105	async with httpx.AsyncClient() as client:
106		response = await client.post(endpoint_url, json={"sid": sid}, timeout=30.0)
107	if response.status_code != 200:
108		print("[ERROR] could not connect data subscription with socketid: ", sid)
109		return
110
111	print("[INFO] client connected with socketid: ", sid)
112

114@_socketio_server.event
115async def disconnect(sid):
116	"""
117	This func is called by the socketio server when a client disconnects.
118	It will trigger an endpoint to trigger the corresponding inner function in the update loop.
119	This complex workaround is necessary, because the socketio server is not part of the fastapi app (it is just mounted
120	to the path), so it cannot access the FastAPI's dependency to the system task queue function to trigger logic within
121	the update loop.
122
123	:param sid: socket id
124	:return:
125	"""
126	print("disconnect")
127	global origin_url
128	if origin_url is None:
129		print(
130			"[ERROR] origin_url is unknown, could not disconnect data subscription with socketid: ",
131			sid,
132		)
133		return
134	endpoint_url = origin_url + "/api/v1/system/data_subscription/disconnect_client"
135	# perform api request with httpx
136	async with httpx.AsyncClient() as client:
137		response = await client.post(endpoint_url, json={"sid": sid})
138	if response.status_code != 200:
139		print("[ERROR] could not disconnect data subscription with socketid: ", sid)
140		return
141	print("[INFO] client disconnected with socketid: ", sid)
