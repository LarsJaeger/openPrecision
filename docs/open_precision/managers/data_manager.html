 16class DataManager:
 17	"""
 18	The DataManager manages data subscriptions and data updates. The goal is to bundle periodic api calls to reduce
 19	computational expenses and network traffic, as well as to move data update work to the server side. These periodic
 20	tasks either check for new data or are otherwise required to be called periodically.
 21
 22	Concretely, equivalent system tasks are grouped, then called once if required by the schedule, then multicasted to
 23	the specified clients via socketio. This is done by the open_precision.api.utils.engine_endpoint decorator, which
 24	also contains more documentation on how data subscriptions work.
 25	"""
 26
 27	def __init__(self, manager: SystemHub):
 28		self.endpoint_dict = None
 29		self._hub = manager
 30		url = "redis://redis:6379"
 31		self._sio = AsyncServer(
 32			client_manager=AsyncRedisManager(url),
 33			async_mode="asgi",
 34			cors_allowed_origins="*",
 35		)
 36		self._data_update_mapping: Dict[
 37			DataSubscription, List[str]
 38		] = {}  # subscription: [subscribed_sids]
 39		self._data_update_mem: Dict[
 40			DataSubscription, Tuple[Any, datetime | None]
 41		] = {}  # subscription: (value, last_updated)
 42		self._subscription_exec_time_mapping: Dict[
 43			DataSubscription, float | None
 44		] = {}  # subscription: average_exec_time
 45		self._connected_clients: list[str] = []

 51		:param sid:
 52		:return:
 53		"""
 54		self._connected_clients.append(sid)

 57		print("connect ", sid)

 64	def inner_on_disconnect(self, sid):
 65		print("inner_disconnect")
 66		print(sid)
 67		print(self._connected_clients)
 68		self._connected_clients.remove(sid)
 69		print(self._connected_clients)
 70		subscriptions_to_delete = []
 71		for key, subscribers_list in self._data_update_mapping.items():
 72			if sid in subscribers_list:
 73				subscribers_list.remove(sid)
 74				if len(subscribers_list) == 0:
 75					subscriptions_to_delete.append(key)
 76		for key in subscriptions_to_delete:
 77			del self._data_update_mapping[key]
 78			del self._data_update_mem[key]
 79			del self._subscription_exec_time_mapping[key]
 80		print("subscriptions_to_delete")
 81		print(subscriptions_to_delete)
 82		print("disconnect ", sid)

 84	async def do_update(self):
 85		"""
 86		This func is called by the SystemHub to trigger a data update. It is not intended to be called from outside
 87		the system update loop. It recomputes out of date data subscriptions and sends an update to subscribers if the
 88		value has changed.
 89		:return: None
 90		"""
 91		# add subscription to out_of_date list if period length has passed and it needs to be updated
 92		out_of_date = []
 93		now = datetime.now()
 94		for subscription, (val, time) in self._data_update_mem.items():
 95			if (time is None) or (
 96				((now - time).total_seconds() * 1000) >= subscription.period_length
 97			):
 98				out_of_date.append(subscription)
 99
100		# if out of date and value changed: send current states to the user interface
101		for subscription in out_of_date:
102			current_mem_val, current_mem_time = self._data_update_mem[subscription]
103			try:
104				time = datetime.now()
105				exec_result = subscription.func(
106					self._hub,
107					*subscription.args,
108					**{x: y for (x, y) in subscription.kw_args},
109				)
110				exec_time = datetime.now() - time
111				# log average execution time per subscription
112				if self._subscription_exec_time_mapping[subscription] is None:
113					self._subscription_exec_time_mapping[
114						subscription
115					] = exec_time.total_seconds()
116				else:
117					self._subscription_exec_time_mapping[subscription] = (
118						self._subscription_exec_time_mapping[subscription] * 0.95
119						+ exec_time.total_seconds() * 0.05
120					)
121				"""
122				print(f"executed data subscription {str(hash(subscription))}")
123				print(f"data_subscription: {subscription}")
124				print(f"result: {exec_result}")
125				"""
126			except Exception:
127				exec_result = {"exception": traceback.format_exc()}
128				if current_mem_time is None or current_mem_val != exec_result:
129					print(
130						f"exception in data subscription {str(hash(subscription))} : \n {traceback.format_exc()}"
131					)
132
133			if current_mem_time is None or current_mem_val != exec_result:
134				self._data_update_mem[subscription] = (exec_result, datetime.now())
135				for subscriber in self._data_update_mapping[subscription]:
136					if isinstance(exec_result, DataModelBase):
137						serialized_result = exec_result.to_json()
138					elif isinstance(exec_result, str):
139						serialized_result = exec_result
140					else:
141						serialized_result = CustomJSONEncoder().encode(exec_result)
142					# print("sending data subscription update to client: ", subscriber)
143					await self._sio.emit(
144						str(hash(subscription)), data=serialized_result, to=subscriber
145					)

147	async def emit_error(self, e: Exception):
148		"""
149		emit an error in the error room
150		"""
151		await self._sio.emit("error", str(e), room="error")

153	def add_data_subscription(self, sid: str, data_subscription: DataSubscription):
154		"""
155		subscribe a socket id to a data subscription
156		:param sid:
157		:param data_subscription:
158		:return:
159		"""
160		if data_subscription in self._data_update_mapping.keys():
161			subscribers = self._data_update_mapping[data_subscription]
162		else:
163			subscribers = []
164			self._data_update_mapping[data_subscription] = subscribers
165			self._data_update_mem[data_subscription] = (None, None)
166			self._subscription_exec_time_mapping[data_subscription] = None
167		if sid not in subscribers:
168			print(f"{sid} subscribed to {str(data_subscription.func)}")
169			subscribers.append(sid)
170
171	def remove_data_subscription(self, sid: str, data_subscription: DataSubscription):
172		"""
173		remove a data subscription for a given socket id and data subscription
174		:param sid:
175		:param data_subscription:
176		:return:
177		"""
178		subscribers = self._data_update_mapping[data_subscription]
179		if sid in subscribers:
180			subscribers.remove(sid)
181			print(f"{sid} unsubscribed from {str(data_subscription.func)}")
182		if len(subscribers) == 0:
183			del self._data_update_mapping[data_subscription]
184			del self._data_update_mem[data_subscription]
185
186	def remove_all_data_subscriptions(self, sid: str):
187		"""
188		remove all data subscriptions for a given socket id
189		:param sid:
190		:return:
191		"""
192		for data_subscription in self._data_update_mapping.keys():
193			self.remove_data_subscription(sid, data_subscription)
194
195	def get_exec_times(self) -> Dict[DataSubscription, float | None]:
196		"""
197		get the average execution times of all data subscriptions
198		:return: dict[DataSubscription, float | None]
199		"""
200		return self._subscription_exec_time_mapping
