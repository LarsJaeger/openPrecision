 16def get_classes_in_package(package, classes: list | None = None) -> list:
 17	if classes is None:
 18		classes = []
 19
 20	# recursively walk the supplied package to retrieve all classes
 21	imported_package = __import__(package, fromlist=["a"])
 22
 23	for _, plugin_name, is_package in pkgutil.iter_modules(
 24		imported_package.__path__, imported_package.__name__ + "."
 25	):
 26		if not is_package:
 27			try:
 28				plugin_module = __import__(plugin_name, fromlist=["a"])
 29				found_classes = inspect.getmembers(plugin_module, inspect.isclass)
 30				for name, cls in found_classes:
 31					if str(cls.__module__) == plugin_name:
 32						classes.append(cls)
 33			except ModuleNotFoundError:
 34				print(f"[INFO] ModuleNotFoundError when importing {plugin_name}")
 35				print(traceback.format_exc())
 36				continue
 37
 38	# Now that we have looked at all the modules in the current package, start looking recursively for additional
 39	# modules in sub packages
 40	all_current_paths = []
 41	if isinstance(imported_package.__path__, str):
 42		all_current_paths.append(imported_package.__path__)
 43	else:
 44		all_current_paths.extend([x for x in imported_package.__path__])
 45
 46	seen_paths = []
 47	for pkg_path in all_current_paths:
 48		if pkg_path not in seen_paths:
 49			seen_paths.append(pkg_path)
 50
 51			# Get all subdirectory of the current package path directory
 52			child_pkgs = [
 53				p
 54				for p in os.listdir(pkg_path)
 55				if os.path.isdir(os.path.join(pkg_path, p))
 56			]
 57
 58			# For each subdirectory, apply the get_classes_in_package method recursively
 59			for child_pkg in child_pkgs:
 60				get_classes_in_package(package + "." + child_pkg, classes)
 61	return classes

 64def _check_plugins_for_class(plugin_class, plugins) -> list:
 65	"""returns all plugins that are a subclass of plugin_class"""
 66
 67	checked_plugins = []
 68	for plugin in plugins:
 69		if issubclass(plugin, plugin_class):
 70			checked_plugins.append(plugin)
 71	return checked_plugins

 74class PluginManager:
 75	def __init__(
 76		self, manager: SystemHub, plugin_type_class: type, plugin_package: str
 77	):
 78		atexit.register(self._cleanup)
 79		self._manager: SystemHub = manager
 80		self._plugin_type_class: type = plugin_type_class
 81		self._plugin_package: str = plugin_package
 82		self._manager.config.register_value(
 83			self, f"loading_priority.{self._plugin_type_class.__name__}", []
 84		)
 85		print(
 86			f"[ClassPluginManager] loading plugins for type: {str(self._plugin_type_class.__name__)}"
 87		)
 88		self._plugin_instance: any = self._initialise_plugin()
 89		print(
 90			f"[ClassPluginManager] initialised plugin: {str(self._plugin_instance.__class__)}"
 91		)
 92		print(
 93			f"[ClassPluginManager] finished loading plugins for type: {str(self.plugin_type_class.__name__)}"
 94		)
 95
 96	def _cleanup(self) -> None:
 97		pass
 98
 99	def _initialise_plugin(self) -> any:
100		# get possible plugins
101		plugins = get_classes_in_package(self._plugin_package, [])
102		possible_plugins = _check_plugins_for_class(self._plugin_type_class, plugins)
103		# get plugin loading priority
104		plugin_loading_priority = self._manager.config.get_value(
105			self, f"loading_priority.{self._plugin_type_class.__name__}"
106		)
107		for plugin in possible_plugins:
108			if plugin.__name__ not in plugin_loading_priority:
109				plugin_loading_priority.append(plugin.__name__)
110		# save back to config
111		self._manager.config.set_value(
112			self,
113			f"loading_priority.{self._plugin_type_class.__name__}",
114			plugin_loading_priority,
115		)
116
117		# create reverse lookup table
118		plugins_from_name = {plugin.__name__: plugin for plugin in possible_plugins}
119
120		# initialises first initialisable class from plugin loading priority
121		if possible_plugins is None:
122			raise MissingPluginException(
123				str(self._plugin_type_class), self._plugin_package
124			)
125		for current_init_plugin in plugin_loading_priority:
126			try:
127				return plugins_from_name[current_init_plugin](self._manager)
128			except PluginException:
129				print(
130					f"[ERROR] Aborting! An error occurred while enabling {current_init_plugin}: {str(PluginException)}"
131				)
132				traceback.print_exc()
133			except:  # noqa
134				print(
135					f"[ERROR] Aborting! An error occurred while enabling {current_init_plugin}."
136				)
137				traceback.print_exc()
