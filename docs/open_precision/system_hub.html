 24class SystemHub:
 25	"""
 26	reponsible for dependency injection, instance management, starting, reloading and stopping the system
 27	"""
 28
 29	def __init__(self):
 30		self._signal_reload = True  # The system will reload if this is True
 31		while self._signal_reload:
 32			# loading sub managers
 33			self._config_path = os.path.join(
 34				os.path.abspath(os.path.dirname(__file__)),
 35				os.path.relpath("../config/config.yml"),
 36			)
 37
 38			self._config = ConfigManager(self)
 39
 40			# map model
 41			print("[INFO]: sleeping for 5 sec")
 42			time.sleep(5)
 43			print("[INFO]: starting model mapping")
 44			self._config.register_value(
 45				self, "neo4j_address", "bolt://neo4j:password@neo4j:7687"
 46			)
 47			map_model(database_url=self._config.get_value(self, "neo4j_address"))
 48			print("[INFO]: finished model mapping")
 49
 50			self._data = DataManager(self)
 51
 52			self._vehicles = VehicleManager(self)
 53
 54			self._system_task_manager = SystemTaskManager(self)
 55
 56			# loading plugins, but loading UserInterface last
 57			self._plugins = {}
 58			for plugin_type in plugin_manager.get_classes_in_package(
 59				"open_precision.core.plugin_base_classes"
 60			):
 61				self._plugins[plugin_type] = PluginManager(
 62					self, plugin_type, "open_precision.plugins"
 63				).instance
 64			self._plugin_name_mapping = _get_plugin_name_mapping(self._plugins)
 65
 66			# initializing and starting api and user interface delivery
 67			self._api = API(self)
 68			print("[INFO]: starting api thread")
 69			with self._api:
 70				# starting update loop
 71				self._signal_stop = False  # the update loop will stop if this is True
 72				print("[INFO]: starting update loop")
 73				asyncio.run(self.start_update_loop())
 74				print("[INFO]: stopped update loop")
 75			print("[INFO]: stopped api thread")
 76
 77	async def start_update_loop(self):
 78		while not self._signal_stop:
 79			artificial_slow_down = asyncio.sleep(0.05)
 80
 81			# handle actions and deliver responses
 82			try:
 83				await self._system_task_manager.handle_tasks(amount=10)
 84			except Exception as e:
 85				print(f"[ERROR]: Error while handling system tasks: {e}")
 86				await self._data.emit_error(e)
 87
 88			# doing data updates (of data subscriptions)
 89			try:
 90				await self._data.do_update()
 91			except Exception as e:
 92				print(f"[ERROR]: Error while handling data updates: {e}")
 93				await self._data.emit_error(e)
 94			await (
 95				artificial_slow_down
 96			)  # uncomment to artificially slow down the update loop
 97
 98	def _stop_update_loop(self):
 99		self._signal_stop = True
100
101	def stop(self):
102		self._signal_reload = False
103		self._stop_update_loop()
