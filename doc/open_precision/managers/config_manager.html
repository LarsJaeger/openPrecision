 17class ConfigManager:
 18	def __init__(self, manager: SystemHub):
 19		self._manager: SystemHub = manager
 20		self._config: CommentedMap = CommentedMap()
 21		self._config_path = self._manager._config_path
 22		self.load_config()
 23		self.classes = plugin_manager.get_classes_in_package("open_precision")
 24		for cls in self.classes:
 25			YAML().register_class(cls)  # register class
 26
 27	def register_value(
 28		self, origin_object: object, key: str, value: any
 29	) -> ConfigManager:
 30		"""adds key/value pair to object's config if not already set"""
 31
 32		address = type(origin_object).__name__
 33		if key is not None:
 34			address += "." + key
 35		flat_conf = flatten(self._config, reducer="dot")
 36		if address not in flat_conf.keys():
 37			flat_conf[address] = value
 38		self._config = CommentedMap(
 39			unflatten(flat_conf, splitter="dot")
 40		)  # update config
 41		return self
 42
 43	def set_value(self, origin_object: object, key: str, value: any) -> ConfigManager:
 44		"""updates key's value in object's config"""
 45		address = type(origin_object).__name__
 46		if key is not None:
 47			address += "." + key
 48		flat_conf = flatten(self._config, reducer="dot")
 49		flat_conf[address] = value
 50		self._config = CommentedMap(unflatten(flat_conf, splitter="dot"))
 51		self.save_config_file()  # TODO possibly cache and save
 52		return self
 53
 54	def get_value(self, origin_object: object, key: str) -> any:
 55		"""returns value of key from origin_object's config"""
 56		address = type(origin_object).__name__
 57		if key is not None:
 58			address += "." + key
 59		flat_conf = flatten(self._config, reducer="dot")
 60		return (
 61			unflatten(flat_conf[address])
 62			if isinstance(flat_conf[address], dict)
 63			else flat_conf[address]
 64		)

 66	def cleanup(self):
 67		self.save_config_file()

 69	def load_config(self, yaml: str = None, reload: bool = False):
 70		"""
 71		loads config file from yaml string or file
 72		:param reload: reinitializing backend after loading config if this is set to True
 73		:param yaml: if None, loads from file, else loads from this parameter (should be the yaml string)
 74		:return: None
 75		"""
 76		if yaml is None:
 77			print("[LOG]: loading config from file")
 78			try:
 79				with open(self._config_path, "r") as config_file_stream:
 80					self._config = YAML().load(stream=config_file_stream)
 81			except FileNotFoundError:
 82				print("[WARNING]: config file not found")
 83				self._config = CommentedMap()
 84		elif isinstance(yaml, str):
 85			print("[LOG]: loading config from string")
 86			self._config = YAML().load(yaml)
 87		else:
 88			raise TypeError("yaml must be None or str")
 89		self._config = CommentedMap() if self._config is None else self._config
 90		print("[DEBUG]: config: ", self._config)
 91		if reload:
 92			self._manager.reload()
 93
 94	def save_config_file(self):
 95		print("[LOG]: saving config file")
 96		try:
 97			print("[DEBUG]: config: ", self._config)
 98			with open(self._config_path, "w") as config_file_stream:
 99				YAML().dump(self._config, stream=config_file_stream)
100		except RepresenterError as e:
101			print("[ERROR]: could not parse config file: ")
102			print(self._config)
103			print(" ".join(traceback.format_exception(e, value=e, tb=e.__traceback__)))
104			return
105
106	def get_config_string(self) -> str:
107		try:
108			config_buffer = io.StringIO()
109			YAML().dump(self._config, stream=config_buffer)
110			return config_buffer.getvalue()
111		except RepresenterError as e:
112			print("[ERROR]: could not parse config file: ")
113			print(self._config)
114			print(" ".join(traceback.format_exception(e, value=e, tb=e.__traceback__)))
115			return ""
